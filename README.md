# C-loops
A series of simple C++ loops

   // while loop example
   // the value of v decreases by 1 until the condition is complete
   
   #include <iostream>
   using namespace std;
   int main()
   {
      int v;
      v = 6;
  
  while(v > 0)
  {
          cout << "v = " << v << endl;
          v--;
  }
  
     return 0;
  }
